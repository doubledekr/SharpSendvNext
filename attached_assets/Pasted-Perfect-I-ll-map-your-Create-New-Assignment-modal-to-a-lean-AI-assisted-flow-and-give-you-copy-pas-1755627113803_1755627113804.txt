Chip input (Key Points) – minimal JS
const points = [];
const input = document.querySelector('#keypointInput');
const hidden = document.querySelector('#assignmentKeypointsValue');
const wrap = document.querySelector('#assignmentKeypoints');

input.addEventListener('keydown', e => {
  if (e.key === 'Enter' && input.value.trim()) {
    e.preventDefault();
    if (points.length >= 3) return alert('Max 3 key points');
    points.push(input.value.trim());
    hidden.value = JSON.stringify(points);
    renderChips();
    input.value = '';
  }
});
function renderChips(){
  wrap.querySelectorAll('.chip').forEach(n=>n.remove());
  points.forEach((p,i)=>{
    const chip=document.createElement('span');
    chip.className='chip';
    chip.textContent=p; 
    chip.onclick=()=>{ points.splice(i,1); hidden.value=JSON.stringify(points); renderChips(); };
    wrap.insertBefore(chip, input);
  });
}
